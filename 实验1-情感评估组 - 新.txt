        // CBC加密：密钥和IV硬编码
        const SECRET_KEY = "travel-assistant-2026-secret-key"; // 长度需满足AES-256 (32字节)
        const SECRET_IV = "random-iv-16byte";               // 16字节 IV
        
        // AES-CBC加密函数
        function encryptAES(data) {
            try {
                const key = CryptoJS.enc.Utf8.parse(SECRET_KEY);
                const iv = CryptoJS.enc.Utf8.parse(SECRET_IV);
                
                const encrypted = CryptoJS.AES.encrypt(
                    data,
                    key,
                    {
                        iv: iv,
                        mode: CryptoJS.mode.CBC,
                        padding: CryptoJS.pad.Pkcs7
                    }
                );
                return encrypted.toString();
            } catch (error) {
                console.error('加密失败:', error);
                return '';
            }
        }
        
        // AES-CBC解密函数
        function decryptAES(encryptedData) {
            try {
                const key = CryptoJS.enc.Utf8.parse(SECRET_KEY);
                const iv = CryptoJS.enc.Utf8.parse(SECRET_IV);
                
                const decrypted = CryptoJS.AES.decrypt(
                    encryptedData,
                    key,
                    {
                        iv: iv,
                        mode: CryptoJS.mode.CBC,
                        padding: CryptoJS.pad.Pkcs7
                    }
                );
                return decrypted.toString(CryptoJS.enc.Utf8);
            } catch (error) {
                console.error('解密失败:', error);
                return null;
            }
        }
        
        // 根据问卷数据生成确认码 - 新格式：反映每个题项选择的数字
        function generateConfirmationCode(feedbackData) {
            console.log('🔢 根据问卷数据生成确认码...');
            
            // 基础确认码
            let baseCode = 'EMOTIONAL-SUPPORT-2026-COMPLETE-CARE';
            
            // 添加问卷答案编码
            let answersCode = '';
            const questionCount = 8; // 总共7个问题
            
            // 检查feedbackData是否有效
            if (!feedbackData || typeof feedbackData !== 'object') {
                console.error('❌ feedbackData无效:', feedbackData);
                throw new Error('问卷数据无效');
            }
            
            console.log('📊 问卷数据内容:', JSON.stringify(feedbackData, null, 2));
            
            // 生成答案编码
            for (let i = 1; i <= questionCount; i++) {
                const questionKey = `q${i}`;
                const answer = feedbackData[questionKey];
                
                console.log(`🔍 问题 ${i} (${questionKey}) 的答案:`, answer);
                
                if (answer === undefined || answer === null || answer === '') {
                    console.warn(`⚠️ 问题 ${i} 没有答案，使用默认值 1`);
                    // 使用默认值 1 作为后备
                    answersCode += `${String.fromCharCode(64 + i)}1`;
                } else {
                    // 使用字母A-T代表20个问题，后面跟用户选择的数字
                    const questionLetter = String.fromCharCode(64 + i); // A=65, B=66, ..., T=84
                    answersCode += `${questionLetter}${answer}`;
                }
            }
            
            // 组合完整的确认码明文
            const plainCode = `${baseCode}-${answersCode}-END`;
            console.log('📝 生成的明文确认码:', plainCode);
            
            // 使用AES-CBC加密确认码
            const encryptedCode = encryptAES(plainCode);
            console.log('🔒 加密后的确认码:', encryptedCode);
            
            // 验证确认码格式
            if (plainCode.length < 50) { // 一个合理的最小长度
                console.warn('⚠️ 生成的确认码可能不完整:', plainCode);
            }
            
            // 返回加密后的确认码
            return encryptedCode;
        }
        
        // 简单的哈希函数
        function simpleHash(str) {
            let hash = 0;
            for (let i = 0; i < str.length; i++) {
                const char = str.charCodeAt(i);
                hash = ((hash << 5) - hash) + char;
                hash = hash & hash; // Convert to 32bit integer
            }
            return Math.abs(hash);
        }

        // 全局复制函数，直接在HTML中调用
        window.copyCodeToClipboard = function(code) {
            console.log('尝试复制确认码:', code);
            
            // 使用现代的 Clipboard API
            if (navigator.clipboard) {
                navigator.clipboard.writeText(code).then(() => {
                    console.log('Clipboard API 复制成功');
                    showCopySuccess();
                }).catch(err => {
                    console.error('Clipboard API 复制失败:', err);
                    // 如果 Clipboard API 失败，尝试传统方法
                    fallbackCopyTextToClipboard(code);
                });
            } else {
                // 如果浏览器不支持 Clipboard API，使用传统方法
                fallbackCopyTextToClipboard(code);
            }
        };

        // 传统的复制方法作为备选
        function fallbackCopyTextToClipboard(text) {
            const textArea = document.createElement('textarea');
            textArea.value = text;
            textArea.style.position = 'fixed';
            textArea.style.left = '-999999px';
            textArea.style.top = '-999999px';
            document.body.appendChild(textArea);
            textArea.focus();
            textArea.select();
            
            try {
                const successful = document.execCommand('copy');
                if (successful) {
                    console.log('传统方法复制成功！');
                    showCopySuccess();
                } else {
                    console.log('传统方法复制失败');
                    alert('复制失败，请手动选择确认码进行复制：\n' + text);
                }
            } catch (err) {
                console.error('复制命令执行失败:', err);
                alert('复制失败，请手动选择确认码进行复制：\n' + text);
            } finally {
                document.body.removeChild(textArea);
            }
        }

        function showNextQuestion() {
            const questionTagsContainer = document.getElementById('question-tags');
            const nextQuestionIndex = conversationState.currentQuestionIndex;
            
            if (nextQuestionIndex < conversationState.totalQuestions) {
                const nextQuestion = dialogueScript[nextQuestionIndex];
                
                // 创建新的问题标签
                const questionTag = document.createElement('button');
                questionTag.className = 'question-tag';
                questionTag.setAttribute('data-question-index', nextQuestionIndex);
                
                // 设置问题文本内容
                questionTag.textContent = nextQuestion.question;
                
                // 绑定点击事件
                questionTag.addEventListener('click', () => {
                    if (!conversationState.isTyping && !questionTag.classList.contains('asked')) {
                        // 检查频率限制
                        const currentTime = Date.now();
                        const timeSinceLastQuestion = currentTime - conversationState.lastQuestionTime;
                        
                        if (conversationState.lastQuestionTime > 0 && timeSinceLastQuestion < conversationState.minQuestionInterval) {
                            showFrequencyWarning();
                            return;
                        }
                        
                        const question = nextQuestion.question;
                        
                        // 更新上次问题时间
                        conversationState.lastQuestionTime = currentTime;
                        
                        // 标记当前问题为已问
                        let userImageUrl = null;
                        

                        
                        handleUserMessage(question, nextQuestionIndex, userImageUrl);
                        questionTag.classList.add('asked');
                        
                        // 隐藏当前问题标签
                        questionTag.classList.add('hidden');
                        
                        // 显示下一个推荐问题（如果还有）
                        if (nextQuestionIndex < conversationState.totalQuestions - 1) {
                            conversationState.currentQuestionIndex = nextQuestionIndex + 1;
                            showNextQuestion();
                        }
                    }
                });
                
                // 添加到容器并显示动画
                questionTagsContainer.appendChild(questionTag);
                gsap.from(questionTag, {
                    x: -20,
                    opacity: 0,
                    duration: 0.3,
                    ease: "power2.out"
                });
            }
        }

                previewContainer.classList.remove('hidden');
                photoUploadArea.classList.add('hidden');
